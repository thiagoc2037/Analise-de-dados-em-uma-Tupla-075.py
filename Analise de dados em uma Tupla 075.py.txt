núm = (int(input('Digite um número: ')),
       int(input('Digite outro número: ')),
       int(input('Digite mais número: ')),
       int(input('Digite o último número: ')))
print(f'Você digitou os valores {núm}')
print(f'O valor 9 apareceu na {núm.count(9)} vezes ')
if 3 in núm:
    print(f'O valor 3 apreceu na {núm.index(3)+1}ª posicão')
else:
    print('O valor 3 não foi digitado em nenhuma posicão')
print('Os valores pares digitados forma ', end='')
for n in núm:
    if n % 2 == 0:
        print(n,end='')
